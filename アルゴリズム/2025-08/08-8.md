n = gets.to_i

total_time = []
n.times do |i|
    start_time, flight_time, arrival_time = gets.split.map(&:to_i)
    if (start_time + flight_time) == arrival_time
        total_time << 24
    else
        total_time << (start_time + flight_time) + (24 - arrival_time) 
    end
end

puts total_time.min
puts total_time.max

### [paizaスキルチェック「C102:行きたいライブのスケジュール」](https://paiza.jp/challenges/share/9kkoet7dSVxLs2VXREV5KzbRwHNnDsX2-RHxdlaF6VY?source=social)
#### 自身の解答
```ruby
a_days = gets.to_i
a_live = []
a_days.times do
    a_live << gets.to_i
end

b_days = gets.to_i
b_live = []
b_days.times do
    b_live << gets.to_i
end
i = 1
dupli = "A"
31.times do
    if a_live.include?(i) && b_live.include?(i)
        if dupli == "A"
            puts dupli
            dupli = "B"
        else
            puts dupli
            dupli = "A"
        end
    elsif a_live.include?(i)
        puts "A"
    elsif b_live.include?(i)
        puts "B"
    else
        puts "x"
    end
    i += 1
end

```
